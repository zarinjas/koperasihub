<script setup>
import { Upload } from 'lucide-vue-next';
import { reactive } from 'vue';

const props = defineProps({
    fields: {
        type: Array,
        required: true,
    },
    model: {
        type: Object,
        required: true,
    },
    errors: {
        type: Object,
        default: () => ({}),
    },
    prefix: {
        type: String,
        required: true,
    },
    idPrefix: {
        type: String,
        default: null,
    },
});

const emit = defineEmits(['update:model']);

const imagePreviews = reactive({});
const inputClass = 'h-11 w-full min-w-0 rounded-lg border border-slate-300 bg-white px-3 text-sm text-slate-950 shadow-sm transition focus:border-teal-700 focus:outline-none focus:ring-2 focus:ring-teal-700/20';
const textareaClass = 'min-h-28 w-full min-w-0 rounded-lg border border-slate-300 bg-white px-3 py-3 text-sm text-slate-950 shadow-sm transition focus:border-teal-700 focus:outline-none focus:ring-2 focus:ring-teal-700/20';

const isFile = (value) => typeof File !== 'undefined' && value instanceof File;
const fieldId = (key) => `${props.idPrefix || props.prefix}-${key}`;
const errorFor = (key) => props.errors[`${props.prefix}.${key}`] || '';
const imagePreviewKey = (field) => `${props.idPrefix || props.prefix}.${field.key}`;
const imagePreviewFor = (field) => imagePreviews[imagePreviewKey(field)] || props.model[`${field.key}_url`] || '';

const normaliseOptions = (options = []) => options.map((option) => (
    option && typeof option === 'object'
        ? option
        : { value: option, label: String(option) }
));

const normaliseValue = (field, value) => {
    if (field.type === 'number') {
        return value === '' ? null : Number(value);
    }

    return value;
};

const updateModel = (nextValues) => {
    emit('update:model', {
        ...(props.model || {}),
        ...nextValues,
    });
};

const setFieldValue = (field, value) => {
    updateModel({
        [field.key]: normaliseValue(field, value),
    });
};

const setImageField = (field, event) => {
    const file = event.target.files?.[0];

    if (!file) {
        return;
    }

    const key = imagePreviewKey(field);

    if (imagePreviews[key]) {
        URL.revokeObjectURL(imagePreviews[key]);
    }

    imagePreviews[key] = URL.createObjectURL(file);
    updateModel({ [field.key]: file });
};

const repeaterItems = (field) => Array.isArray(props.model[field.key]) ? props.model[field.key] : [];

const addRepeaterItem = (field) => {
    const item = Object.fromEntries((field.fields || []).map((nestedField) => [
        nestedField.key,
        nestedField.type === 'toggle' ? false : (nestedField.type === 'image' ? null : ''),
    ]));

    updateModel({
        [field.key]: [...repeaterItems(field), item],
    });
};

const removeRepeaterItem = (field, index) => {
    updateModel({
        [field.key]: repeaterItems(field).filter((_, itemIndex) => itemIndex !== index),
    });
};

const setRepeaterValue = (field, index, nestedField, value) => {
    const items = repeaterItems(field).map((item, itemIndex) => {
        if (itemIndex !== index) {
            return item;
        }

        return {
            ...item,
            [nestedField.key]: normaliseValue(nestedField, value),
        };
    });

    updateModel({ [field.key]: items });
};

const setRepeaterImageField = (field, index, nestedField, event) => {
    const file = event.target.files?.[0];

    if (!file) {
        return;
    }

    const key = `${imagePreviewKey(field)}.${index}.${nestedField.key}`;

    if (imagePreviews[key]) {
        URL.revokeObjectURL(imagePreviews[key]);
    }

    imagePreviews[key] = URL.createObjectURL(file);
    setRepeaterValue(field, index, nestedField, file);
};

const repeaterImagePreviewFor = (field, index, nestedField, item) => {
    const key = `${imagePreviewKey(field)}.${index}.${nestedField.key}`;

    return imagePreviews[key] || item[`${nestedField.key}_url`] || '';
};
</script>

<template>
    <div class="grid gap-4">
        <template v-for="field in fields" :key="`${prefix}-${field.key}`">
            <div v-if="['text', 'url', 'email', 'number'].includes(field.type)" class="w-full space-y-2">
                <label :for="fieldId(field.key)" class="text-sm font-medium text-slate-800">{{ field.label }}</label>
                <input
                    :id="fieldId(field.key)"
                    :type="field.type === 'number' ? 'number' : 'text'"
                    :value="model[field.key] ?? ''"
                    :class="inputClass"
                    :aria-invalid="Boolean(errorFor(field.key))"
                    @input="setFieldValue(field, $event.target.value)"
                />
                <p v-if="field.type === 'url'" class="text-xs leading-5 text-slate-500">Boleh guna URL relatif seperti /tentang-kami atau URL penuh.</p>
                <p v-if="errorFor(field.key)" class="text-sm text-red-700">{{ errorFor(field.key) }}</p>
            </div>

            <div v-else-if="field.type === 'textarea'" class="w-full space-y-2">
                <label :for="fieldId(field.key)" class="text-sm font-medium text-slate-800">{{ field.label }}</label>
                <textarea
                    :id="fieldId(field.key)"
                    :value="model[field.key] ?? ''"
                    :class="textareaClass"
                    :aria-invalid="Boolean(errorFor(field.key))"
                    @input="setFieldValue(field, $event.target.value)"
                />
                <p v-if="field.help" class="text-xs leading-5 text-slate-500">{{ field.help }}</p>
                <p v-if="errorFor(field.key)" class="text-sm text-red-700">{{ errorFor(field.key) }}</p>
            </div>

            <div v-else-if="field.type === 'select'" class="w-full space-y-2">
                <label :for="fieldId(field.key)" class="text-sm font-medium text-slate-800">{{ field.label }}</label>
                <select
                    :id="fieldId(field.key)"
                    :value="model[field.key] ?? ''"
                    :class="inputClass"
                    :aria-invalid="Boolean(errorFor(field.key))"
                    @change="setFieldValue(field, $event.target.value)"
                >
                    <option v-for="option in normaliseOptions(field.options)" :key="option.value" :value="option.value">
                        {{ option.label }}
                    </option>
                </select>
                <p v-if="errorFor(field.key)" class="text-sm text-red-700">{{ errorFor(field.key) }}</p>
            </div>

            <div v-else-if="field.type === 'toggle'" class="flex items-start justify-between gap-4 rounded-2xl border border-slate-200 bg-slate-50 px-4 py-3">
                <div>
                    <p class="text-sm font-medium text-slate-900">{{ field.label }}</p>
                    <p v-if="field.help" class="mt-1 text-xs leading-5 text-slate-500">{{ field.help }}</p>
                </div>
                <button
                    type="button"
                    class="relative inline-flex h-6 w-11 shrink-0 items-center rounded-full transition"
                    :class="model[field.key] ? 'bg-teal-700' : 'bg-slate-300'"
                    @click="setFieldValue(field, !model[field.key])"
                >
                    <span
                        class="inline-block h-5 w-5 transform rounded-full bg-white shadow transition"
                        :class="model[field.key] ? 'translate-x-5' : 'translate-x-1'"
                    />
                </button>
            </div>

            <div v-else-if="field.type === 'image'" class="space-y-3">
                <label class="text-sm font-medium text-slate-800">{{ field.label }}</label>
                <label
                    :for="fieldId(field.key)"
                    class="flex cursor-pointer flex-col items-center justify-center gap-3 rounded-2xl border border-dashed border-slate-300 bg-slate-50 px-4 py-8 text-center transition hover:border-teal-300 hover:bg-teal-50/40"
                >
                    <span class="flex h-12 w-12 items-center justify-center rounded-2xl bg-white text-teal-700 shadow-sm">
                        <Upload class="h-5 w-5" />
                    </span>
                    <div class="space-y-1">
                        <p v-if="isFile(model[field.key])" class="text-sm font-medium text-slate-900">{{ model[field.key].name }}</p>
                        <p v-else class="text-sm font-medium text-slate-900">Pilih fail imej untuk dimuat naik</p>
                        <p class="text-xs leading-5 text-slate-500">JPEG, PNG, JPG atau WebP. Maks 2MB.</p>
                    </div>
                </label>
                <input
                    :id="fieldId(field.key)"
                    accept="image/jpeg,image/png,image/jpg,image/webp"
                    type="file"
                    class="sr-only"
                    @change="setImageField(field, $event)"
                />
                <div v-if="imagePreviewFor(field)" class="overflow-hidden rounded-2xl border border-slate-200 bg-white p-2 shadow-sm">
                    <img
                        :src="imagePreviewFor(field)"
                        class="h-44 w-full rounded-xl object-cover"
                        :alt="isFile(model[field.key]) ? 'Pratonton imej baharu' : 'Imej semasa'"
                    />
                    <p class="mt-2 text-xs text-slate-500">
                        {{ isFile(model[field.key]) ? 'Pratonton imej baharu. Tekan Simpan untuk mengemas kini laman.' : 'Imej semasa.' }}
                    </p>
                </div>
                <p v-if="errorFor(field.key)" class="text-sm text-red-700">{{ errorFor(field.key) }}</p>
            </div>

            <div v-else-if="field.type === 'repeater'" class="space-y-4 rounded-2xl border border-slate-200 bg-slate-50 p-4">
                <div class="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
                    <div>
                        <h3 class="text-sm font-semibold text-slate-900">{{ field.label }}</h3>
                        <p v-if="field.max_items" class="text-xs leading-5 text-slate-500">Maksimum {{ field.max_items }} item.</p>
                    </div>
                    <button type="button" class="inline-flex h-10 items-center justify-center rounded-lg border border-slate-200 bg-white px-4 text-sm font-medium text-slate-700 shadow-sm transition-colors hover:bg-slate-100" @click="addRepeaterItem(field)">
                        Tambah Item
                    </button>
                </div>

                <div v-if="errors[`${prefix}.${field.key}`]" class="text-sm text-red-700">
                    {{ errors[`${prefix}.${field.key}`] }}
                </div>

                <div v-if="repeaterItems(field).length === 0" class="rounded-2xl border border-dashed border-slate-300 bg-white p-4 text-sm text-slate-500">
                    Belum ada item ditambah.
                </div>

                <div v-for="(item, index) in repeaterItems(field)" :key="`${field.key}-${index}`" class="rounded-2xl border border-slate-200 bg-white p-4 shadow-sm">
                    <div class="mb-4 flex items-center justify-between">
                        <p class="text-sm font-semibold text-slate-900">Item {{ index + 1 }}</p>
                        <button type="button" class="rounded-lg px-3 py-2 text-sm font-medium text-red-600 hover:bg-red-50 hover:text-red-700" @click="removeRepeaterItem(field, index)">
                            Padam
                        </button>
                    </div>

                    <div class="grid gap-4">
                        <template v-for="nestedField in field.fields" :key="`${field.key}-${index}-${nestedField.key}`">
                            <div v-if="['text', 'url', 'email', 'number'].includes(nestedField.type)" class="w-full space-y-2">
                                <label :for="fieldId(`${field.key}-${index}-${nestedField.key}`)" class="text-sm font-medium text-slate-800">{{ nestedField.label }}</label>
                                <input
                                    :id="fieldId(`${field.key}-${index}-${nestedField.key}`)"
                                    :type="nestedField.type === 'number' ? 'number' : 'text'"
                                    :value="item[nestedField.key] ?? ''"
                                    :class="inputClass"
                                    :aria-invalid="Boolean(errors[`${prefix}.${field.key}.${index}.${nestedField.key}`])"
                                    @input="setRepeaterValue(field, index, nestedField, $event.target.value)"
                                />
                                <p v-if="errors[`${prefix}.${field.key}.${index}.${nestedField.key}`]" class="text-sm text-red-700">
                                    {{ errors[`${prefix}.${field.key}.${index}.${nestedField.key}`] }}
                                </p>
                            </div>

                            <div v-else-if="nestedField.type === 'image'" class="space-y-3">
                                <label class="text-sm font-medium text-slate-800">{{ nestedField.label }}</label>
                                <label
                                    :for="fieldId(`${field.key}-${index}-${nestedField.key}`)"
                                    class="flex cursor-pointer flex-col items-center justify-center gap-3 rounded-2xl border border-dashed border-slate-300 bg-slate-50 px-4 py-8 text-center transition hover:border-teal-300 hover:bg-teal-50/40"
                                >
                                    <span class="flex h-12 w-12 items-center justify-center rounded-2xl bg-white text-teal-700 shadow-sm">
                                        <Upload class="h-5 w-5" />
                                    </span>
                                    <div class="space-y-1">
                                        <p v-if="isFile(item[nestedField.key])" class="text-sm font-medium text-slate-900">{{ item[nestedField.key].name }}</p>
                                        <p v-else class="text-sm font-medium text-slate-900">Pilih fail imej untuk item ini</p>
                                        <p class="text-xs leading-5 text-slate-500">JPEG, PNG, JPG atau WebP. Maks 2MB.</p>
                                    </div>
                                </label>
                                <input
                                    :id="fieldId(`${field.key}-${index}-${nestedField.key}`)"
                                    accept="image/jpeg,image/png,image/jpg,image/webp"
                                    type="file"
                                    class="sr-only"
                                    @change="setRepeaterImageField(field, index, nestedField, $event)"
                                />
                                <div v-if="repeaterImagePreviewFor(field, index, nestedField, item)" class="overflow-hidden rounded-2xl border border-slate-200 bg-white p-2 shadow-sm">
                                    <img
                                        :src="repeaterImagePreviewFor(field, index, nestedField, item)"
                                        class="h-44 w-full rounded-xl object-cover"
                                        :alt="isFile(item[nestedField.key]) ? 'Pratonton imej baharu' : 'Imej semasa'"
                                    />
                                    <p class="mt-2 text-xs text-slate-500">
                                        {{ isFile(item[nestedField.key]) ? 'Pratonton imej baharu. Tekan Simpan untuk mengemas kini laman.' : 'Imej semasa.' }}
                                    </p>
                                </div>
                                <p v-if="errors[`${prefix}.${field.key}.${index}.${nestedField.key}`]" class="text-sm text-red-700">
                                    {{ errors[`${prefix}.${field.key}.${index}.${nestedField.key}`] }}
                                </p>
                            </div>

                            <div v-else class="w-full space-y-2">
                                <label :for="fieldId(`${field.key}-${index}-${nestedField.key}`)" class="text-sm font-medium text-slate-800">{{ nestedField.label }}</label>
                                <textarea
                                    :id="fieldId(`${field.key}-${index}-${nestedField.key}`)"
                                    :value="item[nestedField.key] ?? ''"
                                    :class="textareaClass"
                                    :aria-invalid="Boolean(errors[`${prefix}.${field.key}.${index}.${nestedField.key}`])"
                                    @input="setRepeaterValue(field, index, nestedField, $event.target.value)"
                                />
                                <p v-if="errors[`${prefix}.${field.key}.${index}.${nestedField.key}`]" class="text-sm text-red-700">
                                    {{ errors[`${prefix}.${field.key}.${index}.${nestedField.key}`] }}
                                </p>
                            </div>
                        </template>
                    </div>
                </div>
            </div>
        </template>
    </div>
</template>
