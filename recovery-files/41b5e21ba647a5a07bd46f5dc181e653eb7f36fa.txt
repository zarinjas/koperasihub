<script setup>
import { Head, Link } from '@inertiajs/vue3';
import { computed } from 'vue';
import AdminLayout from '@/Admin/Layouts/AdminLayout.vue';
import { Button } from '@/Shared/Components/ui/button';
import { Mail, Pencil } from 'lucide-vue-next';

const props = defineProps({
    templates: { type: Object, required: true },
    templateTypes: { type: Object, required: true },
    canEdit: { type: Boolean, default: false },
});

const templateList = computed(() => {
    return Object.entries(props.templateTypes).map(([type, meta]) => ({
        type,
        label: meta.label,
        variables: meta.variables,
        template: props.templates[type] ?? null,
    }));
});

const placeholder = (v) => '\x7b\x7b' + v + '\x7d\x7d';
</script>

<template>
    <Head title="Templat E-mel" />

    <AdminLayout>
        <div class="space-y-6">
            <div class="flex flex-col gap-3 sm:flex-row sm:items-end sm:justify-between">
                <div>
                    <h1 class="text-2xl font-semibold tracking-normal">Templat E-mel</h1>
                    <p class="mt-1 max-w-2xl text-sm leading-6 text-slate-600">
                        Urus kandungan e-mel yang dihantar secara automatik oleh sistem.
                    </p>
                </div>
            </div>

            <div class="grid gap-4 md:grid-cols-2">
                <div
                    v-for="item in templateList"
                    :key="item.type"
                    class="flex flex-col gap-4 rounded-2xl border border-slate-200 bg-white p-5 shadow-sm"
                >
                    <div class="flex items-start justify-between gap-3">
                        <div class="flex items-start gap-3">
                            <span class="flex h-10 w-10 shrink-0 items-center justify-center rounded-xl bg-teal-50 text-teal-700">
                                <Mail class="h-5 w-5" />
                            </span>
                            <div class="min-w-0">
                                <h2 class="font-semibold text-slate-950">{{ item.label }}</h2>
                                <p class="mt-0.5 text-xs text-slate-500">
                                    {{ item.template ? item.template.subject : 'Guna templat lalai sistem' }}
                                </p>
                            </div>
                        </div>
                        <span v-if="item.template?.is_active" class="inline-flex items-center rounded-full border border-emerald-200 bg-emerald-50 px-2.5 py-0.5 text-xs font-medium text-emerald-700">
                            Aktif
                        </span>
                        <span v-else class="inline-flex items-center rounded-full border border-slate-200 bg-slate-50 px-2.5 py-0.5 text-xs font-medium text-slate-500">
                            Lalai
                        </span>
                    </div>

                    <div v-if="item.variables.length" class="flex flex-wrap gap-1">
                        <span class="text-xs text-slate-500">Pemboleh ubah:</span>
                        <code
                            v-for="v in item.variables"
                            :key="v"
                            class="rounded-md bg-slate-100 px-1.5 py-0.5 font-mono text-xs text-slate-700"
                        >{{ placeholder(v) }}</code>
                    </div>

                    <div class="mt-auto flex justify-end border-t border-slate-100 pt-3">
                        <Button
                            :as="Link"
                            :href="'/admin/email-templates/' + item.type + '/edit'"
                            variant="outline"
                            size="sm"
                            :disabled="!canEdit"
                        >
                            <Pencil class="mr-1.5 h-3.5 w-3.5" />
                            Edit
                        </Button>
                    </div>
                </div>
            </div>
        </div>
    </AdminLayout>
</template>
