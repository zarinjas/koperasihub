export function getFieldTypeConfig(value) {
  return FIELD_TYPES.find((t) => t.value === value);
}

export function getCategoryFields(categoryKey) {
  return FIELD_TYPES.filter((t) => t.category === categoryKey);
}

export function searchFieldTypes(query) {
  const q = query.toLowerCase().trim();
  if (!q) return FIELD_TYPES;
  return FIELD_TYPES.filter((t) => {
    if (t.label.toLowerCase().includes(q)) return true;
    if (t.description.toLowerCase().includes(q)) return true;
    if (t.keywords.some((k) => k.includes(q))) return true;
    if (t.value.toLowerCase().includes(q)) return true;
    return false;
  });
}
